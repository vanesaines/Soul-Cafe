// ########################################
// como definir variables
// ########################################
var a = 1;
let a = 1;
const a = 1;

// var es global
{
    var a = 1;
}
a += 1;  // funciona

// let vale solo en el scope en el que esta definido
{
    let a = 1;
}
a += 1;  // ERROR

// const es como let, pero es una constante. No se puede reasignar.
{
    const a = 1;
}
a += 1;  // ERROR, a es constante
